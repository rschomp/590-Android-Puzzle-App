    public void oneButtonClicked(View v){ 
        for (int m = 0; m<clickedList.length; m++) { 
            Button butM = clickedList[m]; 
            butM.performClick(); 
 
            if (boardFull()) { 
                String fullSeqPhrase = "Sequence: " + getTheTag(butM); 
                seq.setText(fullSeqPhrase); 
                mc.setText("Move Count: 1"); 
                return; 
            } else { 
                butM.performClick(); 
            }
